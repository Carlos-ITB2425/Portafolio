/
SOURCE_DIR="/home/usuario/datos_compartidos"       
BACKUP_DIR="/mnt/backup/servidor/"                
LOG_FILE="/var/log/backup.log"                   
DATE=$(date +%Y-%m-%d_%H-%M)


mkdir -p "$BACKUP_DIR"


rsync -avz --delete "$SOURCE_DIR" "$BACKUP_DIR/Backup_$DATE" >> "$LOG_FILE" 2>&1

# Verificar el resultado
if [ $? -eq 0 ]; then
    echo "$DATE - Backup completado exitosamente." >> "$LOG_FILE"
else
    echo "$DATE - Error en el backup." >> "$LOG_FILE"
fi

crontab -e

0 2 * * * /bin/bash /ruta/al/backup.sh
